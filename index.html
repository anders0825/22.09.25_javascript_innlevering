function getGreeting(name, time) {
  let greeting = "";

  if (time >= 0 && time <= 5) {
    greeting = "God Natt";
  } else if (time >= 6 && time <= 11) {
    greeting = "God morgen";
  } else if (time >= 12 && time <= 17) {
    greeting = "God dag";
  } else if (time >= 18 && time <= 23) {
    greeting = "God Kveld";

  if ((time < 0) | (time > 23)) {
    return `Ugyldig Tid`;

}

function opSyvD(arr, str) {
  if (arr.includes(str)) {
    let index = arr.indexOf(str);
    arr.splice(index, 1);
  } else {
    arr.push(str);
  }
  return arr;
}
